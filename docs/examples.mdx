## Multiple modal

<Playground>
  {() => {
    const [openFirst, setOpenFirst] = React.useState(false);
    const [openSecond, setOpenSecond] = React.useState(false);

    const littleLorem = (
      <p>
        Lorem ipsum dolor sit amet, consectetur adipiscing elit. Nullam pulvinar
        risus non risus hendrerit venenatis. Pellentesque sit amet hendrerit
        risus, sed porttitor quam.
      </p>
    );

    return (
      <div>
        <button onClick={() => setOpenFirst(true)}>
          Open first modal
        </button>

        <Modal open={openFirst} onClose={() => setOpenFirst(false)} center>
          <p>First modal</p>
          {littleLorem}
          <button onClick={() => setOpenSecond(true)}>
            Open second modal
          </button>
        </Modal>

        <Modal open={openSecond} onClose={() => setOpenSecond(false)} center>
          <p>Second modal</p>
          {littleLorem}
        </Modal>
      </div>
    );

}}

</Playground>

## Custom container

<Playground>
  {() => {
    const [open, setOpen] = React.useState(false);
    const myRef = React.useRef();

    return (
      <div>
        <div ref={myRef} />

        <button onClick={() => setOpen(true)}>
          Open modal
        </button>

        <Modal open={open} onClose={() => setOpen(false)} center container={myRef.current}>
          <p>
            Take a look with the devtools you will see that this modal is
            rendered in a custom container.
          </p>
        </Modal>
      </div>
    );

}}

</Playground>
